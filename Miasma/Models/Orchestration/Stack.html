def load_data(args={})
  args = args.to_smash
  @resources = (args.delete(:resources) || []).each do |r|
    Resource.new(r)
  end
  @events = (args.delete(:events) || []).each do |e|
    Event.new(e)
  end
  super args
end

def reload
  clear_memoizations!
  remove = data.keys.find_all do |k|
    ![:id, :name].include?(k.to_sym)
  end
  remove.each do |k|
    data.delete(k)
  end
  super
end
