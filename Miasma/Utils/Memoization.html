def clear_memoizations!
  _memo.keys.find_all do |key|
    key.to_s.start_with?("#{self.object_id}_")
  end.each do |key|
    _memo.delete(key)
  end
  true
end

def memoize(key, direct=false)
  unless(direct)
    key = "#{self.object_id}_#{key}"
  end
  unless(_memo.has_key?(key))
    _memo[key] = yield
  end
  _memo[key]
end
