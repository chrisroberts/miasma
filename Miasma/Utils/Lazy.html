def included(klass)
  klass.class_eval do
    include InstanceMethods
    extend ClassMethods

    class << self

      def inherited(klass)
        klass.set_attributes(
          MultiJson.load(
            MultiJson.dump(self.attributes)
          ).to_smash
        )
      end

    end
  end
end
