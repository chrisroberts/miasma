def model(klass=nil)
  if(klass)
    unless(klass.ancestors.include?(Miasma::Types::Model))
      raise TypeError.new "Expecting `Miasma::Types::Model` subclass! (got #{klass})"
    else
      self._model = klass
    end
  end
  self._model
end
