def initialize(api, model_data=nil)
  @api = api
  @data = Smash.new
  @dirty = Smash.new
  if(model_data)
    if(model_data.is_a?(Hash))
      load_data(model_data) unless model_data.empty?
    else
      raise TypeError.new "Expecting `model_data` to be of type `Hash`. Received: `#{model_data.class}`"
    end
  end
end
